% Slide 3: Reinforcement Learning Method
\begin{frame}{Reinforcement Learning Method: PPO Algorithm}
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{PPO Objective Function:}
\begin{equation*}
L^{CLIP}(\theta) = \mathbb{E}_t\left[\min\left(r_t(\theta)\hat{A}_t, \text{clip}(r_t(\theta), 1-\epsilon, 1+\epsilon)\hat{A}_t\right)\right]
\end{equation*}

\textbf{Observation Space (Flat): 48-dim}
\begin{itemize}
    \item Body linear/angular velocity (6-dim)
    \item Projected gravity (3-dim)
    \item Joint positions/velocities (24-dim)
    \item Previous actions (12-dim)
    \item Velocity command (3-dim)
\end{itemize}

\textbf{Action Space: 12-dim}
\begin{itemize}
    \item Desired joint positions (PD controller input)
\end{itemize}

\column{0.45\textwidth}
\textbf{Reward Function Design:}
\begin{itemize}
    \item \textbf{Velocity tracking} (weight 3.0)
    \begin{equation*}
    \small r_{lin} = 3.0 \cdot e^{-\frac{\|v_{cmd} - v_{actual}\|^2}{0.15}}
    \end{equation*}

    \item \textbf{Upright orientation} (weight 0.5)
    \item \textbf{Smooth motion} (weight -0.05)
    \item \textbf{Foot clearance} (weight -0.75)
    \item \textbf{Gait timing} (weight 0.25)
\end{itemize}

\vspace{0.2cm}
\textbf{Domain Randomization:}
\begin{itemize}
    \item Friction coefficient: $\mu \in [0.5, 1.2]$
    \item Mass: $\pm 20\%$
    \item Motor strength: $\pm 10\%$
\end{itemize}
\end{columns}
\end{frame}

% Slide 5: Critical Finding - Domain Randomization
\begin{frame}{Critical Finding: Domain Randomization is Essential}
\begin{center}
\includegraphics[width=0.9\textwidth]{dr_comparison.png}
\end{center}

\begin{block}{Critical Discovery}
\textbf{Without domain randomization, the policy completely fails to learn locomotion!} The robot either remains stationary or falls immediately.
\end{block}

\textbf{Why DR is Essential:}
\begin{itemize}
    \item Prevents overfitting to nominal parameters (friction=1.0, nominal mass)
    \item Forces policy to discover robust strategies that work across conditions
    \item Prepares for sim-to-real transfer (real-world parameters never match simulation exactly)
\end{itemize}
\end{frame}

% Slide 7: Challenges and Solutions
\begin{frame}{Challenges and Solutions}
\begin{block}{Challenge 1: Training Instability}
\textbf{Problem:} High reward variance in early training, policy fails to converge\\
\textbf{Solution:} Introduced domain randomization, adjusted reward weights, increased parallel environments
\end{block}

\begin{block}{Challenge 2: Poor Lateral Motion Performance}
\textbf{Problem:} Low lateral velocity tracking (84.1\%), 0\% push recovery rate\\
\textbf{Plan:} Increase lateral motion reward weight, extend training time, introduce explicit disturbance training
\end{block}

\begin{block}{Challenge 3: Low Energy Efficiency}
\textbf{Problem:} High CoT value (133 vs target <100)\\
\textbf{Plan:} Increase smooth motion penalty weight, optimize gait patterns
\end{block}

\begin{block}{Challenge 4: High Observation Dimensionality}
\textbf{Problem:} Slope terrain requires 187-dim height scan, slowing training\\
\textbf{Solution:} Reduced parallel environments (512 vs 2048), used more efficient network architecture
\end{block}
\end{frame}
